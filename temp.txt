"use client";

import { useCallback, useEffect, useMemo, useState } from "react";
import { Loader2 } from "lucide-react";
import { useForm } from "react-hook-form";
import { zodResolver } from "@hookform/resolvers/zod";
import { Button } from "@/components/ui/button";
import { Input } from "@/components/ui/input";
import { Label } from "@/components/ui/label";
import { cn } from "@/lib/utils";
import { PasswordStrength } from "@/components/auth2/PasswordStrength";
import { EmailAvailabilityHint } from "@/components/auth2/EmailAvailabilityHint";
import { useToast } from "@/components/ui/toast";
import { useDebounce } from "@/hooks/useDebounce";
import {
  EmailSchema,
  SignupStep1Schema,
  type SignupRole,
  type SignupStep1Input,
} from "@/lib/auth/schemas";
import { getPasswordStrength, getPasswordStrengthLabel } from "@/lib/validation";

const STORAGE_KEY = "signup_user_data";
const roleOptions: Array<{ value: SignupRole; title: string; description: string }> = [
  {
    value: "creator",
    title: "Createur",
    description:
      "Participez aux concours, envoyez vos contenus et suivez votre classement en direct.",
  },
  {
    value: "brand",
    title: "Marque",
    description:
      "Deployez vos campagnes UGC, analysez les performances et recompensez les meilleurs talents.",
  },
];

type PasswordStrengthLevel = "weak" | "medium" | "strong";

interface Step1AccountCreationProps {
  onContinue?: (data: SignupStep1Input) => void;
}

export function Step1AccountCreation({ onContinue }: Step1AccountCreationProps) {
  const { toast } = useToast();

  const {
    register,
    handleSubmit,
    watch,
    setValue,
    setError,
    clearErrors,
    formState: { errors, isSubmitting },
  } = useForm<SignupStep1Input>({
    resolver: zodResolver(SignupStep1Schema),
    mode: "onBlur",
    defaultValues: {
      email: "",
      password: "",
      confirmPassword: "",
      role: "creator",
    },
  });

  const roleValue = (watch("role") as SignupRole | undefined) ?? "creator";
  const emailValue = watch("email") ?? "";
  const passwordValue = watch("password") ?? "";

  const debouncedEmail = useDebounce(emailValue.trim(), 400);
  const debouncedPassword = useDebounce(passwordValue, 300);

  const [emailAvailability, setEmailAvailability] = useState<boolean | null>(null);
  const [isCheckingEmail, setIsCheckingEmail] = useState(false);
  const [emailFailureReason, setEmailFailureReason] = useState<"offline" | "error" | null>(
    null,
  );

  const passwordStrengthScore = useMemo(
    () => getPasswordStrength(debouncedPassword),
    [debouncedPassword],
  );

  const normalizedPasswordScore = useMemo(() => {
    if (!debouncedPassword) return 0;
    if (passwordStrengthScore >= 70) return 4;
    if (passwordStrengthScore >= 50) return 3;
    if (passwordStrengthScore >= 30) return 2;
    if (passwordStrengthScore > 0) return 1;
    return 0;
  }, [debouncedPassword, passwordStrengthScore]);

  const passwordStrength: PasswordStrengthLevel = useMemo(() => {
    if (normalizedPasswordScore >= 4) return "strong";
    if (normalizedPasswordScore >= 2) return "medium";
    return "weak";
  }, [normalizedPasswordScore]);

  const passwordStrengthLabel = useMemo(
    () => getPasswordStrengthLabel(passwordStrengthScore),
    [passwordStrengthScore],
  );

  useEffect(() => {
    if (!debouncedEmail) {
      setEmailAvailability(null);
      setEmailFailureReason(null);
      setIsCheckingEmail(false);
      clearErrors("email");
      return;
    }

    const parsed = EmailSchema.safeParse(debouncedEmail);
    if (!parsed.success) {
      setEmailAvailability(null);
      setEmailFailureReason(null);
      clearErrors("email");
      return;
    }

    if (typeof navigator !== "undefined" && navigator?.onLine === false) {
      setEmailAvailability(null);
      setIsCheckingEmail(false);
      setEmailFailureReason("offline");
      setError("email", {
        type: "manual",
        message: "Vérification indisponible — connexion perdue.",
      });
      return;
    }

    const controller = new AbortController();
    setIsCheckingEmail(true);
    setEmailFailureReason(null);

    (async () => {
      try {
        const response = await fetch("/api/auth/check-email", {
          method: "POST",
          headers: {
            "Content-Type": "application/json",
          },
          body: JSON.stringify({ email: debouncedEmail }),
          signal: controller.signal,
        });

        if (!response.ok) {
          throw new Error("verification_failed");
        }

        const data = (await response.json().catch(() => null)) as
          | { available?: boolean }
          | null;
        const available =
          typeof data?.available === "boolean" ? data?.available : null;

        if (available === null) {
          throw new Error("invalid_response");
        }

        setEmailAvailability(available);
        setEmailFailureReason(null);

        if (available) {
          clearErrors("email");
        } else {
          setError("email", {
            type: "manual",
            message: "Adresse déjà utilisée ✕",
          });
        }
      } catch (error) {
        if (controller.signal.aborted) return;
        setEmailAvailability(null);
        const offline =
          typeof navigator !== "undefined" && navigator?.onLine === false;
        const reason = offline ? "offline" : "error";

        setEmailFailureReason(reason);
        setError("email", {
          type: "manual",
          message:
            reason === "offline"
              ? "Vérification indisponible — connexion perdue."
              : "Vérification indisponible — réessayez.",
        });
  }, [debouncedEmail, clearErrors, setError]);

  const emailCheckFailed = emailFailureReason !== null;

  const disableSubmit =
    isSubmitting || isCheckingEmail || emailAvailability === false || emailCheckFailed;

  const emailHintStatus = useMemo(() => {
    if (emailCheckFailed) return "unavailable";
    if (!emailValue) return "idle";
    if (isCheckingEmail) return "checking";
    if (emailAvailability === false) return "unavailable";
    if (emailAvailability === true) return "available";
    return "idle";
  }, [emailAvailability, emailCheckFailed, emailValue, isCheckingEmail]);

  const emailHintMessage = useMemo(() => {
    if (emailFailureReason === "offline") return "Vérification indisponible — connexion perdue.";
    if (emailFailureReason === "error") return "Vérification indisponible — réessayez.";
    if (emailHintStatus === "unavailable") return "Adresse déjà utilisée ✕";
    if (emailHintStatus === "available") return "Adresse disponible ✓";
    if (emailHintStatus === "checking") return "Verification en cours...";
    return undefined;
  }, [emailFailureReason, emailHintStatus]);

  const persistSignupData = useCallback((email: string, role: SignupRole) => {
    if (typeof window === "undefined") return;
    try {
      const payload = {
        email,
        role,
        timestamp: Date.now(),
      };
      window.localStorage.setItem(STORAGE_KEY, JSON.stringify(payload));
    } catch (error) {
      console.warn("Impossible de stocker les donnees signup:", error);
    }
  }, []);

  const handleRoleChange = useCallback(
    (nextRole: SignupRole) => {
      if (roleValue !== nextRole) {
        setValue("role", nextRole, { shouldDirty: true, shouldValidate: false });
      }
    },
    [roleValue, setValue],
  );

  const onSubmit = useCallback(
    async (values: SignupStep1Input) => {
      const payload = {
        email: values.email,
        password: values.password,
        confirmPassword: values.confirmPassword,
        role: values.role,
      };

      try {
        const response = await fetch("/api/auth/signup", {
          method: "POST",
          headers: {
            "Content-Type": "application/json",
          },
          body: JSON.stringify(payload),
        });

        const result = await response.json().catch(() => null);

        if (!response.ok || !result?.success) {
          const message =
            typeof result?.error === "string"
              ? result.error
              : "Impossible de creer le compte. Verifiez les informations saisies.";

          toast({
            title: "Creation impossible",
            description: message,
            variant: "error",
          });
          return;
        }

        persistSignupData(values.email, values.role);

        toast({
          title: "Verifiez votre email",
          description: "Un lien de confirmation vient de vous etre envoye.",
        });

        onContinue?.(values);
      } catch (error) {
        const message =
          error instanceof Error
            ? error.message
            : "Erreur inattendue lors de la creation du compte.";
        toast({
          title: "Creation impossible",
          description: message,
          variant: "error",
        });
      }
    },
    [onContinue, persistSignupData, toast],
  );

  return (
    <div aria-live="polite" className="space-y-8">
      <div className="space-y-2 text-left">
        <h2 className="text-xl font-semibold text-zinc-900 dark:text-zinc-100">
          Creez votre acces
        </h2>
        <p className="text-sm text-zinc-600 dark:text-zinc-400">
          Selectionnez votre profil, renseignez vos identifiants et continuez.
        </p>
      </div>

      <div className="grid gap-3 sm:grid-cols-2" role="radiogroup" aria-label="Choisissez un role">
        {roleOptions.map((role) => {
          const selected = roleValue === role.value;
          return (
            <button
              key={role.value}
              type="button"
              onClick={() => handleRoleChange(role.value)}
              className={cn(
                "group flex w-full flex-col rounded-2xl border px-5 py-4 text-left transition focus-visible:outline-none focus-visible:ring-2 focus-visible:ring-indigo-500 focus-visible:ring-offset-2 focus-visible:ring-offset-white dark:focus-visible:ring-offset-zinc-950",
                selected
                  ? "border-indigo-400 bg-indigo-50 text-indigo-900 shadow-sm dark:border-indigo-500/70 dark:bg-indigo-500/10 dark:text-indigo-200"
                  : "border-zinc-200 bg-white hover:border-indigo-200 hover:bg-indigo-50/40 dark:border-zinc-800 dark:bg-zinc-900 dark:hover:border-indigo-800/70 dark:hover:bg-indigo-500/5",
              )}
              role="radio"
              aria-checked={selected}
              aria-pressed={selected}
            >
              <span className="text-sm font-semibold">{role.title}</span>
              <span className="mt-1 text-xs text-zinc-600 dark:text-zinc-400">
                {role.description}
              </span>
            </button>
          );
        })}
      </div>

      <form className="space-y-6" onSubmit={handleSubmit(onSubmit)} noValidate>
        <div className="space-y-2">
          <Label htmlFor="signup-email">Adresse email professionnelle</Label>
          <Input
            id="signup-email"
            type="email"
            placeholder="vous@exemple.com"
            autoComplete="email"
            aria-invalid={errors.email ? "true" : "false"}
            {...register("email")}
          />
          <EmailAvailabilityHint status={emailHintStatus} message={emailHintMessage} />
          {errors.email ? (
            <p className="text-sm text-rose-600 dark:text-rose-400" role="alert">
              {errors.email.message}
            </p>
          ) : null}
        </div>

        <div className="grid gap-4 sm:grid-cols-2">
          <div className="space-y-2">
            <Label htmlFor="signup-password">Mot de passe</Label>
            <Input
              id="signup-password"
              type="password"
              placeholder="Creer un mot de passe"
              autoComplete="new-password"
              aria-invalid={errors.password ? "true" : "false"}
              {...register("password")}
            />
            {errors.password ? (
              <p className="text-sm text-rose-600 dark:text-rose-400" role="alert">
                {errors.password.message}
              </p>
            ) : null}
          </div>
          <div className="space-y-2">
            <Label htmlFor="signup-confirm">Confirmer le mot de passe</Label>
            <Input
              id="signup-confirm"
              type="password"
              placeholder="Confirmer"
              autoComplete="new-password"
              aria-invalid={errors.confirmPassword ? "true" : "false"}
              {...register("confirmPassword")}
            />
            {errors.confirmPassword ? (
              <p className="text-sm text-rose-600 dark:text-rose-400" role="alert">
                {errors.confirmPassword.message}
              </p>
            ) : null}
          </div>
        </div>

        <div className="space-y-1">
          <PasswordStrength score={normalizedPasswordScore} strength={passwordStrength} />
          <p className="text-xs text-zinc-500 dark:text-zinc-400" aria-live="polite">
            Force du mot de passe : {passwordStrengthLabel}
          </p>
        </div>

        <input type="hidden" readOnly value={roleValue} {...register("role")} />

        <div className="space-y-2 text-center">
          <Button
            type="submit"
            className="w-full sm:w-auto"
            aria-label="Continuer vers la verification email"
            disabled={disableSubmit}
            aria-busy={isSubmitting}
            title={
              isSubmitting
                ? "Creation en cours..."
                : emailCheckFailed
                  ? "Verification email indisponible."
                  : undefined
            }
          >
            {isSubmitting ? (
              <>
                <Loader2 className="mr-2 h-4 w-4 animate-spin" aria-hidden="true" />
                Creation...
              </>
            ) : (
              "Continuer"
            )}
          </Button>

          {isCheckingEmail ? (
            <p className="text-xs text-zinc-500 dark:text-zinc-400" aria-live="polite">
              Verification de disponibilite en cours...
            </p>
          ) : null}
        </div>

        <p className="text-center text-sm text-zinc-600 dark:text-zinc-400">
          Deja inscrit(e) ?{" "}
          <a
            href="/login"
            className="font-medium text-indigo-600 underline-offset-4 transition hover:underline dark:text-indigo-400"
          >
            Connexion
          </a>
        </p>
      </form>
    </div>
  );
}
